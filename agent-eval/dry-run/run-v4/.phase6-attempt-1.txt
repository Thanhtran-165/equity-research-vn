<section id="sec-glossary" class="section">
  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Thuật ngữ tài chính sử dụng trong báo cáo.</div>
  <div class="glossary-grid aos">
    <div class="gloss-item"><div class="gloss-term">P/E (Price-to-Earnings)</div><div class="gloss-def">Giá / EPS. Đo lường giá trả cho 1 đồng lợi nhuận. P/E thấp = rẻ (nếu LN bền vững).</div></div>
    <div class="gloss-item"><div class="gloss-term">P/B (Price-to-Book)</div><div class="gloss-def">Giá / BVPS. Đo lường giá so với giá trị sổ sách. P/B < 1 = rẻ hơn giá trị tài sản.</div></div>
    <div class="gloss-item"><div class="gloss-term">ROE (Return on Equity)</div><div class="gloss-def">Lợi nhuận / Vốn chủ sở hữu. Hiệu suất sinh lời trên vốn cổ đông. >15% = tốt.</div></div>
    <div class="gloss-item"><div class="gloss-term">CFO (Cash Flow from Operations)</div><div class="gloss-def">Dòng tiền từ hoạt động kinh doanh. Dương = tiền thực về, Âm = tiền đang nằm trong dự án.</div></div>
    <div class="gloss-item"><div class="gloss-term">Backlog</div><div class="gloss-def">Tổng giá trị hợp đồng đã ký nhưng chưa ghi nhận doanh thu. Đảm bảo doanh thu tương lai.</div></div>
    <div class="gloss-item"><div class="gloss-term">D/E (Debt-to-Equity)</div><div class="gloss-def">Nợ vay / Vốn chủ sở hữu. Đo lường đòn bẩy. <1.5× = khỏe cho ngành xây dựng.</div></div>
    <div class="gloss-item"><div class="gloss-term">EPS (Earnings Per Share)</div><div class="gloss-def">Lợi nhuận / Số cổ phiếu. LN trên mỗi cổ phiếu. CTD EPS FY2025: 17,400 VND.</div></div>
    <div class="gloss-item"><div class="gloss-term">Split-adjusted</div><div class="gloss-def">Giá đã điều chỉnh theo các đợt phát hành/chia tách. Giúp so sánh lịch sử công bằng.</div></div>
    <div class="gloss-item"><div class="gloss-term">EPC</div><div class="gloss-def">Engineering-Procurement-Construction. Tổng thầu trọn gói từ thiết kế đến thi công.</div></div>
    <div class="gloss-item"><div class="gloss-term">RSI (Relative Strength Index)</div><div class="gloss-def">Chỉ báo động lượng 0-100. >70 = overbought, <30 = oversold.</div></div>
  </div>

function grad(ctx, c1, c2){
  const chart = ctx.chart;
  const {ctx: cc, chartArea} = chart;
  if (!chartArea) return c1;
  const g = cc.createLinearGradient(0, chartArea.top, 0, chartArea.bottom);
  g.addColorStop(0, c1); g.addColorStop(1, c2);
  return g;
}
