Lý giải: NPAT (1,510) - ΔVCSH (-508) = ~2,018 tỷ phải đi đâu đó → gần như chắc chắn là <strong>cổ tức/chia sẻ lợi nhuận rất lớn</strong> (~2,000 tỷ, tương đương ~18,000 VND/CP). Nếu sai (có thể do dữ liệu chưa đầy đủ), cần cross-check với BCTC kiểm toán FY2025.</div>

<section id="sec-glossary" class="section">
  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Thuật ngữ tài chính dùng trong báo cáo.</div>
  <div class="glossary-grid aos">
    <div class="gloss-item"><div class="gloss-term">P/E (Price/Earnings)</div><div class="gloss-def">Giá / EPS. P/E 3.91× = trả 3.91 VND cho mỗi 1 VND lợi nhuận/năm.</div></div>
    <div class="gloss-item"><div class="gloss-term">P/B (Price/Book)</div><div class="gloss-def">Giá / BVPS. P/B 1.0× = mua đúng giá book value.</div></div>
    <div class="gloss-item"><div class="gloss-term">EPS</div><div class="gloss-def">Lợi nhuận trên mỗi cổ phiếu (VND).</div></div>
    <div class="gloss-item"><div class="gloss-term">ROE</div><div class="gloss-def">Lợi nhuận / Vốn chủ sở hữu. ROE 19% = 1 đồng vốn tạo 0.19 đồng lợi nhuận/năm.</div></div>
    <div class="gloss-item"><div class="gloss-term">Back-log</div><div class="gloss-def">Tổng giá trị hợp đồng chưa hoàn thành. Đảm bảo doanh thu tương lai.</div></div>
    <div class="gloss-item"><div class="gloss-term">Bẫy 5B</div><div class="gloss-def">Cổ phiếu rẻ (P/E thấp) nhưng giá không tăng vì lợi nhuận không bền hoặc thị trường định giá thấp có lý do.</div></div>
    <div class="gloss-item"><div class="gloss-term">Split-adjusted</div><div class="gloss-def">Dữ liệu đã điều chỉnh cho các đợt chia cổ phiếu để so sánh chính xác theo thời gian.</div></div>
    <div class="gloss-item"><div class="gloss-term">Cross-check</div><div class="gloss-def">Đối chiếu nhiều nguồn độc lập để xác nhận số liệu.</div></div>
    <div class="gloss-item"><div class="gloss-term">CFO</div><div class="gloss-def">Dòng tiền từ hoạt động kinh doanh. CFO dương = tiền thật vào công ty.</div></div>
    <div class="gloss-item"><div class="gloss-term">Capex</div><div class="gloss-def">Chi phí vốn — đầu tư tài sản cố định. Capex thấp = asset-light.</div></div>
  </div>
</section>

(function(){
  const nav = document.getElementById('topnav');
  const links = Array.from(document.querySelectorAll('.topnav-link'));
  const progressBar = document.getElementById('progressBar');
  const backTop = document.getElementById('backTop');
  const sections = links.map(l => document.getElementById(l.dataset.target)).filter(Boolean);

  function onScroll(){
    const y = window.scrollY;
    const h = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (h>0 ? (y/h*100) : 0) + '%';
    if (y > 600) backTop.classList.add('visible'); else backTop.classList.remove('visible');
    const navH = nav.offsetHeight + 10;
    let active = sections[0];
    for (const s of sections){
      if (s && (s.getBoundingClientRect().top - navH) <= 0) active = s;
    }
    links.forEach(l => {
      l.classList.toggle('active', active && l.dataset.target === active.id);
    });
    const activeLink = links.find(l => l.classList.contains('active'));
    if (activeLink){
      const navInner = nav.querySelector('.topnav-inner');
      const linkRect = activeLink.getBoundingClientRect();
      const innerRect = navInner.getBoundingClientRect();
      if (linkRect.right > innerRect.right - 20 || linkRect.left < innerRect.left + 20){
        navInner.scrollTo({ left: navInner.scrollLeft + (linkRect.left - innerRect.left - 40), behavior:'smooth' });
      }
    }
  }
  window.addEventListener('scroll', onScroll, {passive:true});
  window.addEventListener('resize', onScroll);
  links.forEach(l => {
    l.addEventListener('click', (e)=>{
      const t = document.getElementById(l.dataset.target);
      if (!t) return;
      e.preventDefault();
      const top = t.getBoundingClientRect().top + window.scrollY - nav.offsetHeight - 8;
      window.scrollTo({ top, behavior:'smooth' });
    });
  });
  backTop.addEventListener('click', ()=> window.scrollTo({top:0, behavior:'smooth'}));
  onScroll();
})();
</script>
