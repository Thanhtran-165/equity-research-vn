<section id="sec-glossary" class="section">
  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Thuật ngữ tài chính sử dụng trong báo cáo.</div>
  <div class="glossary-grid aos">
    <div class="gloss-item">
      <div class="gloss-term">P/E (Price-to-Earnings)</div>
      <div class="gloss-def">Giá / EPS. Cho biết bạn trả bao nhiêu cho 1 đồng lợi nhuận. P/E thấp = rẻ (nhưng có thể có lý do).</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">P/B (Price-to-Book)</div>
      <div class="gloss-def">Giá / Giá trị sổ sách. P/B = 1 nghĩa là giá = vốn chủ sở hữu/cổ phiếu.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">EPS</div>
      <div class="gloss-def">Lợi nhuận trên mỗi cổ phiếu. EPS cao = công ty tạo nhiều lợi nhuận/cổ phiếu.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">ROE</div>
      <div class="gloss-def">Lợi nhuận / Vốn chủ sở hữu. Đo hiệu quả sử dụng vốn. ROE >15% là tốt.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">CFO (Cash from Operations)</div>
      <div class="gloss-def">Dòng tiền từ hoạt động kinh doanh. CFO < NPAT = dấu hiệu cảnh báo chất lượng lợi nhuận.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Capex</div>
      <div class="gloss-def">Chi phí vốn (mua tài sản cố định). Capex thấp = mô hình nhẹ tài sản.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Backlog</div>
      <div class="gloss-def">Tổng giá trị hợp đồng chưa hoàn thành. Backlog lớn = doanh thu tương lai đảm bảo.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Drawdown</div>
      <div class="gloss-def">Khoảng giảm từ đỉnh giá. Drawdown -40% = mất 40% từ đỉnh.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">RSI (Relative Strength Index)</div>
      <div class="gloss-def">Chỉ số dao động 0-100. RSI >70 = quá mua, <30 = quá bán.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Bẫy 5F</div>
      <div class="gloss-def">Fad (mốt), Float (thanh khoản), Future (tương lai), Financials (báo cáo), Fraud (lừa đảo) — 5 rủi ro cần kiểm tra.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Split-adjusted</div>
      <div class="gloss-def">Dữ liệu đã điều chỉnh cho các sự kiện chia tách cổ phiếu để so sánh chính xác.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Cross-check</div>
      <div class="gloss-def">Kiểm tra chéo nhiều nguồn độc lập để xác nhận số liệu.</div>
    </div>
  </div>
</section>

function grad(ctx, c1, c2){
  const chart = ctx.chart;
  const {ctx: cc, chartArea} = chart;
  if (!chartArea) return c1;
  const g = cc.createLinearGradient(0, chartArea.top, 0, chartArea.bottom);
  g.addColorStop(0, c1); g.addColorStop(1, c2);
  return g;
}
