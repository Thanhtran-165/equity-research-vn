Nếu không có cổ tức hoặc phát hành mới, vốn CHSH <em>phải tăng</em> thêm 1.510 tỷ. Sự chênh lệch ~2.018 tỷ cần được giải thích.</p>
    <div class="grid-2" style="margin-top:16px">
      <div class="callout info"><div class="callout-title">Giả thuyết 1: Cổ tức lớn</div><div class="callout-body">CTD có thể đã chi trả cổ tức ~2.000 tỷ (tương đương ~18.000 ₫/CP — payout ratio >130%). Điều này sẽ giải thích sự sụt giảm vốn CHSH. Nếu đúng, đây là <strong>tín hiệu tích cực</strong> về dòng tiền và minh bạch.</div></div>
      <div class="callout warn"><div class="callout-title">Giả thuyết 2: Ghi giảm tài sản</div><div class="callout-body">CTD có thể đã trích lập dự phòng hoặc ghi giảm nợ phải thu/tài sản. Nếu đúng, đây là <strong>dọn dẹp BCTC</strong> — tích cực dài hạn nhưng tiêu cực ngắn hạn.</div></div>
    </div>
    <div class="callout plain" style="margin-top:16px"><div class="callout-title">💡 Hành động</div><div class="callout-body">Đọc <strong>thuyết minh BCTC FY2025</strong> phần "Biến động vốn chủ sở hữu" để xác nhận nguyên nhân. Đây là <strong>biến số quan trọng nhất</strong> ảnh hưởng đến đánh giá chất lượng lợi nhuận của CTD.</div></div>

  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub aos">Thuật ngữ tài chính trong báo cáo.</div>
  <div class="glossary-grid aos">
    <div class="gloss-item"><div class="gloss-term">P/E (Price-to-Earnings)</div><div class="gloss-def">Giá / EPS. Cho biết bạn trả bao nhiêu năm lợi nhuận cho 1 cổ phiếu. P/E thấp = rẻ (nhưng có thể rẻ vì lý do).</div></div>
    <div class="gloss-item"><div class="gloss-term">P/B (Price-to-Book)</div><div class="gloss-def">Giá / BVPS. So sánh giá với giá trị sổ sách. P/B &lt; 1 = rẻ hơn giá trị tài sản ròng.</div></div>
    <div class="gloss-item"><div class="gloss-term">EPS</div><div class="gloss-def">Lợi nhuận trên mỗi cổ phiếu (VND). = LNST / Số CP.</div></div>
    <div class="gloss-item"><div class="gloss-term">ROE</div><div class="gloss-def">Lợi nhuận / Vốn CHSH. Đo hiệu quả sử dụng vốn. &gt;15% tốt.</div></div>
    <div class="gloss-item"><div class="gloss-term">Capex</div><div class="gloss-def">Chi phí vốn (mua tài sản cố định, thiết bị). Capex thấp = asset-light.</div></div>
    <div class="gloss-item"><div class="gloss-term">FCF (Free Cash Flow)</div><div class="gloss-def">CFO - Capex. Dòng tiền tự do sau khi đầu tư. Dương = tốt.</div></div>
    <div class="gloss-item"><div class="gloss-term">RSI</div><div class="gloss-def">Chỉ số sức mạnh tương đối. &gt;70 = quá mua, &lt;30 = quá bán.</div></div>
    <div class="gloss-item"><div class="gloss-term">Drawdown</div><div class="gloss-def">Khoảng giảm từ đỉnh. Đo rủi ro downside.</div></div>
  </div>

function grad(ctx, c1, c2){
  const chart = ctx.chart;
  const {ctx: cc, chartArea} = chart;
  if (!chartArea) return c1;
  const g = cc.createLinearGradient(0, chartArea.top, 0, chartArea.bottom);
  g.addColorStop(0, c1); g.addColorStop(1, c2);
  return g;
}

(function(){
  const nav = document.getElementById('topnav');
  const links = Array.from(document.querySelectorAll('.topnav-link'));
  const progressBar = document.getElementById('progressBar');
  const backTop = document.getElementById('backTop');
  const sections = links.map(l => document.getElementById(l.dataset.target)).filter(Boolean);

  function onScroll(){
    const y = window.scrollY;
    const h = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (h>0 ? (y/h*100) : 0) + '%';
    if (y > 600) backTop.classList.add('visible'); else backTop.classList.remove('visible');
    const navH = nav.offsetHeight + 10;
    let active = sections[0];
    for (const s of sections){
      if (s && (s.getBoundingClientRect().top - navH) <= 0) active = s;
    }
    links.forEach(l => {
      l.classList.toggle('active', active && l.dataset.target === active.id);
    });
    const activeLink = links.find(l => l.classList.contains('active'));
    if (activeLink){
      const navInner = nav.querySelector('.topnav-inner');
      const linkRect = activeLink.getBoundingClientRect();
      const innerRect = navInner.getBoundingClientRect();
      if (linkRect.right > innerRect.right - 20 || linkRect.left < innerRect.left + 20){
        navInner.scrollTo({ left: navInner.scrollLeft + (linkRect.left - innerRect.left - 40), behavior:'smooth' });
      }
    }
  }
  window.addEventListener('scroll', onScroll, {passive:true});
  window.addEventListener('resize', onScroll);
  links.forEach(l => {
    l.addEventListener('click', (e)=>{
      const t = document.getElementById(l.dataset.target);
      if (!t) return;
      e.preventDefault();
      const top = t.getBoundingClientRect().top + window.scrollY - nav.offsetHeight - 8;
      window.scrollTo({ top, behavior:'smooth' });
    });
  });
  backTop.addEventListener('click', ()=> window.scrollTo({top:0, behavior:'smooth'}));
  onScroll();
})();
</script>
