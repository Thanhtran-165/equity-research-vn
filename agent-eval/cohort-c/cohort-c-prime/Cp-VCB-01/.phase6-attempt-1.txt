<section id="sec-glossary" class="section">
  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Thuật ngữ tài chính & ngân hàng dùng trong báo cáo.</div>
  <div class="aos">
    <div class="glossary-grid">
      <div class="gloss-item"><div class="gloss-term">P/E (Price/Earnings)</div><div class="gloss-def">Giá / EPS. Định giá dựa trên lợi nhuận. P/E thấp = rẻ (nếu lợi nhuận ổn định).</div></div>
      <div class="gloss-item"><div class="gloss-term">P/B (Price/Book)</div><div class="gloss-def">Giá / giá trị sổ sách. Định giá ngân hàng thường dùng P/B. P/B < 1 = rẻ hơn giá trị sổ sách.</div></div>
      <div class="gloss-item"><div class="gloss-term">ROE (Return on Equity)</div><div class="gloss-def">Lợi nhuận / vốn CSH. Đo hiệu quả sử dụng vốn. ROE > 15% = tốt cho ngân hàng.</div></div>
      <div class="gloss-item"><div class="gloss-term">NIM (Net Interest Margin)</div><div class="gloss-def">Biên lãi thuần = (Thu nhập lãi - Chi phí vốn) / Tín dụng trung bình. NIM cao = hiệu quả cho vay.</div></div>
      <div class="gloss-item"><div class="gloss-term">CASA Ratio</div><div class="gloss-def">Tiền gửi không kỳ hạn / Tổng tiền gửi. CASA cao = chi phí vốn thấp = NIM cao.</div></div>
      <div class="gloss-item"><div class="gloss-term">NPL (Non-Performing Loan)</div><div class="gloss-def">Tỷ lệ nợ xấu. NPL < 1% = an toàn. NPL > 3% = rủi ro cao.</div></div>
      <div class="gloss-item"><div class="gloss-term">CAR (Capital Adequacy Ratio)</div><div class="gloss-def">Tỷ lệ an toàn vốn. CAR > 8% = đạt chuẩn Basel II. VCB có CAR ~11%+.</div></div>
      <div class="gloss-item"><div class="gloss-term">Split-adjusted</div><div class="gloss-def">Điều chỉnh giá/EPS lịch sử theo tỷ lệ chia cổ phiếu để so sánh chính xác.</div></div>
      <div class="gloss-item"><div class="gloss-term">Bẫy 5F</div><div class="gloss-def">Khung 5 yếu tố cảnh báo nợ xấu: Flow, Forward, Foreclosure, Foreign, Franchise.</div></div>
      <div class="gloss-item"><div class="gloss-term">Cross-check</div><div class="gloss-def">Đối chiếu nhiều nguồn để xác minh dữ liệu. Kỷ luật quan trọng khi đọc báo cáo AI.</div></div>

(function(){
  const nav = document.getElementById('topnav');
  const links = Array.from(document.querySelectorAll('.topnav-link'));
  const progressBar = document.getElementById('progressBar');
  const backTop = document.getElementById('backTop');
  const sections = links.map(l => document.getElementById(l.dataset.target)).filter(Boolean);

  function onScroll(){
    const y = window.scrollY;
    const h = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (h>0 ? (y/h*100) : 0) + '%';
    if (y > 600) backTop.classList.add('visible'); else backTop.classList.remove('visible');
    const navH = nav.offsetHeight + 10;
    let active = sections[0];
    for (const s of sections){
      if (s && (s.getBoundingClientRect().top - navH) <= 0) active = s;
    }
    links.forEach(l => {
      l.classList.toggle('active', active && l.dataset.target === active.id);
    });
    const activeLink = links.find(l => l.classList.contains('active'));
    if (activeLink){
      const navInner = nav.querySelector('.topnav-inner');
      const linkRect = activeLink.getBoundingClientRect();
      const innerRect = navInner.getBoundingClientRect();
      if (linkRect.right > innerRect.right - 20 || linkRect.left < innerRect.left + 20){
        navInner.scrollTo({ left: navInner.scrollLeft + (linkRect.left - innerRect.left - 40), behavior:'smooth' });
      }
    }
  }
  window.addEventListener('scroll', onScroll, {passive:true});
  window.addEventListener('resize', onScroll);
  links.forEach(l => {
    l.addEventListener('click', (e)=>{
      const t = document.getElementById(l.dataset.target);
      if (!t) return;
      e.preventDefault();
      const top = t.getBoundingClientRect().top + window.scrollY - nav.offsetHeight - 8;
      window.scrollTo({ top, behavior:'smooth' });
    });
  });
  backTop.addEventListener('click', ()=> window.scrollTo({top:0, behavior:'smooth'}));
  onScroll();
})();
</script>
