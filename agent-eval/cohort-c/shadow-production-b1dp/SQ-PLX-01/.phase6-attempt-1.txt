<section id="sec-glossary" class="section">
  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Các thuật ngữ tài chính sử dụng trong báo cáo.</div>
  <div class="glossary-grid aos">
    <div class="gloss-item">
      <div class="gloss-term">P/E (Price-to-Earnings)</div>
      <div class="gloss-def">Giá / EPS. Cho biết bạn trả bao nhiêu cho 1 đồng lợi nhuận. P/E 12.87× = trả 12.87 đồng cho 1 đồng LN.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">P/B (Price-to-Book)</div>
      <div class="gloss-def">Giá / BVPS. So sánh giá cổ phiếu với giá trị sổ sách. P/B 2.10× = giá gấp 2.1 lần giá trị sổ sách.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">EPS (Earnings Per Share)</div>
      <div class="gloss-def">Lợi nhuận trên mỗi cổ phiếu. PLX 2025: 2,859 ₫/cổ phiếu.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">ROE (Return on Equity)</div>
      <div class="gloss-def">Lợi nhuận / Vốn chủ sở hữu. Đo hiệu quả sử dụng vốn. PLX: ~18%.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Capex (Capital Expenditure)</div>
      <div class="gloss-def">Chi phí vốn — đầu tư tài sản dài hạn (trạm xăng, đường ống). PLX 2025: 2,332 tỷ.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Free Cash Flow (FCF)</div>
      <div class="gloss-def">Dòng tiền hoạt động - Capex. Tiền thực tế công ty tạo ra sau khi đầu tư.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Drawdown</div>
      <div class="gloss-def">Phần trăm sụt giảm từ đỉnh giá cao nhất. Đo rủi ro biến động giá.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">RSI (Relative Strength Index)</div>
      <div class="gloss-def">Chỉ báo động lượng 0–100. >70: quá mua (có thể giảm). <30: quá bán (có thể tăng).</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Split-adjusted</div>
      <div class="gloss-def">Giá đã điều chỉnh cho các đợt chia cổ phiếu để so sánh chính xác theo thời gian.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Cross-check</div>
      <div class="gloss-def">Đối chiếu nhiều nguồn để xác minh số liệu. Kỷ luật quan trọng khi đọc báo cáo AI.</div>
    </div>
  </div>
</section>

function grad(ctx, c1, c2){
  const chart = ctx.chart;
  const {ctx: cc, chartArea} = chart;
  if (!chartArea) return c1;
  const g = cc.createLinearGradient(0, chartArea.top, 0, chartArea.bottom);
  g.addColorStop(0, c1); g.addColorStop(1, c2);
  return g;
}
