<section id="sec-glossary" class="section">
  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Các thuật ngữ tài chính sử dụng trong báo cáo.</div>
  <div class="glossary-grid aos">
    <div class="gloss-item">
      <div class="gloss-term">P/E (Price-to-Earnings)</div>
      <div class="gloss-def">Giá / EPS. Cho biết bạn trả bao nhiêu cho mỗi đồng lợi nhuận. P/E thấp = định giá rẻ (nhưng có thể có lý do).</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">P/B (Price-to-Book)</div>
      <div class="gloss-def">Giá / BVPS. Cho biết bạn trả bao nhiêu trên giá trị sổ sách. P/B cao = thị trường kỳ vọng ROE cao.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">EPS (Earnings Per Share)</div>
      <div class="gloss-def">Lợi nhuận trên mỗi cổ phiếu. Lợi nhuận sau thuế chia cho số cổ phiếu lưu hành.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">ROE (Return on Equity)</div>
      <div class="gloss-def">Lợi nhuận / Vốn chủ sở hữu. Đo lường hiệu quả sử dụng vốn. ROE ≥15% được coi là tốt.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">NPL (Non-Performing Loan)</div>
      <div class="gloss-def">Tỷ lệ nợ xấu. NPL thấp = chất lượng tài sản tốt. NPL &lt;1% là rất tốt cho ngân hàng.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">CASA</div>
      <div class="gloss-def">Tỷ lệ tiền gửi không kỳ hạn. CASA cao = chi phí vốn thấp = lợi thế cạnh tranh.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">NIM (Net Interest Margin)</div>
      <div class="gloss-def">Biên lợi nhuận từ lãi. (Thu nhập từ lãi - Chi phí vốn) / Tổng tài sản sinh lãi.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">BVPS (Book Value Per Share)</div>
      <div class="gloss-def">Vốn chủ sở hữu / Số cổ phiếu. Giá trị sổ sách trên mỗi cổ phiếu.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Split-adjusted</div>
      <div class="gloss-def">Dữ liệu đã được điều chỉnh cho các lần chia tách cổ phiếu, giúp so sánh lịch sử chính xác.</div>
    </div>
    <div class="gloss-item">
      <div class="gloss-term">Bẫy 5B (Value Trap)</div>
      <div class="gloss-def">Cổ phiếu có vẻ rẻ (P/E thấp) nhưng giá tiếp tục giảm vì lý do cơ bản xấu.</div>
    </div>
  </div>

(function(){
  const nav = document.getElementById('topnav');
  const links = Array.from(document.querySelectorAll('.topnav-link'));
  const progressBar = document.getElementById('progressBar');
  const backTop = document.getElementById('backTop');
  const sections = links.map(l => document.getElementById(l.dataset.target)).filter(Boolean);

  function onScroll(){
    const y = window.scrollY;
    const h = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (h>0 ? (y/h*100) : 0) + '%';
    if (y > 600) backTop.classList.add('visible'); else backTop.classList.remove('visible');
    const navH = nav.offsetHeight + 10;
    let active = sections[0];
    for (const s of sections){
      if (s && (s.getBoundingClientRect().top - navH) <= 0) active = s;
    }
    links.forEach(l => {
      l.classList.toggle('active', active && l.dataset.target === active.id);
    });
    const activeLink = links.find(l => l.classList.contains('active'));
    if (activeLink){
      const navInner = nav.querySelector('.topnav-inner');
      const linkRect = activeLink.getBoundingClientRect();
      const innerRect = navInner.getBoundingClientRect();
      if (linkRect.right > innerRect.right - 20 || linkRect.left < innerRect.left + 20){
        navInner.scrollTo({ left: navInner.scrollLeft + (linkRect.left - innerRect.left - 40), behavior:'smooth' });
      }
    }
  }
  window.addEventListener('scroll', onScroll, {passive:true});
  window.addEventListener('resize', onScroll);
  links.forEach(l => {
    l.addEventListener('click', (e)=>{
      const t = document.getElementById(l.dataset.target);
      if (!t) return;
      e.preventDefault();
      const top = t.getBoundingClientRect().top + window.scrollY - nav.offsetHeight - 8;
      window.scrollTo({ top, behavior:'smooth' });
    });
  });
  backTop.addEventListener('click', ()=> window.scrollTo({top:0, behavior:'smooth'}));
  onScroll();
})();
</script>
