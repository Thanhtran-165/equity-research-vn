Khác biệt của BIDV so với ngân hàng tư nhân là <strong>thế mạnh cho vay doanh nghiệp lớn và dự án trọng điểm quốc gia</strong>.</div>

  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Các thuật ngữ tài chính dùng trong báo cáo.</div>
  <div class="glossary-grid aos">
    <div class="gloss-item"><div class="gloss-term">P/E (Price/Earnings)</div><div class="gloss-def">Giá cổ phiếu / EPS. Cho biết bạn trả bao nhiêu cho 1 đồng lợi nhuận. P/E 25× = trả 25 đồng cho 1 đồng lợi nhuận.</div></div>
    <div class="gloss-item"><div class="gloss-term">P/B (Price/Book)</div><div class="gloss-def">Giá cổ phiếu / giá trị sổ sách. P/B 3× = giá gấp 3 lần vốn CSH. Phổ biến cho định giá ngân hàng.</div></div>
    <div class="gloss-item"><div class="gloss-term">EPS</div><div class="gloss-def">Lợi nhuận trên mỗi cổ phiếu = Lợi nhuận sau thuế / số cổ phiếu.</div></div>
    <div class="gloss-item"><div class="gloss-term">ROE</div><div class="gloss-def">Lợi nhuận / Vốn chủ sở hữu. Đo hiệu quả sử dụng vốn. ROE &gt;15% là tốt cho ngân hàng.</div></div>
    <div class="gloss-item"><div class="gloss-term">NIM</div><div class="gloss-def">Net Interest Margin — chênh lệch giữa lãi thu và lãi trả / tài sản sinh lãi. NIM cao = hiệu quả cho vay tốt.</div></div>
    <div class="gloss-item"><div class="gloss-term">NPL</div><div class="gloss-def">Non-Performing Loan — tỷ lệ nợ xấu. NPL &lt;2% là an toàn cho ngân hàng VN.</div></div>
    <div class="gloss-item"><div class="gloss-term">CAR</div><div class="gloss-def">Capital Adequacy Ratio — tỷ lệ an toàn vốn. CAR &gt;8% theo Basel II/III.</div></div>
    <div class="gloss-item"><div class="gloss-term">CASA</div><div class="gloss-def">Current Account Savings Account — tiền gửi không kỳ hạn (giá rẻ). CASA cao = chi phí vốn thấp.</div></div>
    <div class="gloss-item"><div class="gloss-term">Big 4</div><div class="gloss-def">4 ngân hàng TMCP Nhà nước lớn nhất VN: BIDV, Vietcombank, VietinBank, Agribank.</div></div>
    <div class="gloss-item"><div class="gloss-term">Split-adjusted</div><div class="gloss-def">Số liệu đã được điều chỉnh theo chia/cắt cổ phiếu để so sánh chính xác qua thời gian.</div></div>

(function(){
  const nav = document.getElementById('topnav');
  const links = Array.from(document.querySelectorAll('.topnav-link'));
  const progressBar = document.getElementById('progressBar');
  const backTop = document.getElementById('backTop');
  const sections = links.map(l => document.getElementById(l.dataset.target)).filter(Boolean);

  function onScroll(){
    const y = window.scrollY;
    const h = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (h>0 ? (y/h*100) : 0) + '%';
    if (y > 600) backTop.classList.add('visible'); else backTop.classList.remove('visible');
    const navH = nav.offsetHeight + 10;
    let active = sections[0];
    for (const s of sections){
      if (s && (s.getBoundingClientRect().top - navH) <= 0) active = s;
    }
    links.forEach(l => {
      l.classList.toggle('active', active && l.dataset.target === active.id);
    });
    const activeLink = links.find(l => l.classList.contains('active'));
    if (activeLink){
      const navInner = nav.querySelector('.topnav-inner');
      const linkRect = activeLink.getBoundingClientRect();
      const innerRect = navInner.getBoundingClientRect();
      if (linkRect.right > innerRect.right - 20 || linkRect.left < innerRect.left + 20){
        navInner.scrollTo({ left: navInner.scrollLeft + (linkRect.left - innerRect.left - 40), behavior:'smooth' });
      }
    }
  }
  window.addEventListener('scroll', onScroll, {passive:true});
  window.addEventListener('resize', onScroll);
  links.forEach(l => {
    l.addEventListener('click', (e)=>{
      const t = document.getElementById(l.dataset.target);
      if (!t) return;
      e.preventDefault();
      const top = t.getBoundingClientRect().top + window.scrollY - nav.offsetHeight - 8;
      window.scrollTo({ top, behavior:'smooth' });
    });
  });
  backTop.addEventListener('click', ()=> window.scrollTo({top:0, behavior:'smooth'}));
  onScroll();
})();
</script>
