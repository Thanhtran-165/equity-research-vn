  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>

    <div class="glossary-grid">
      <div class="gloss-item"><div class="gloss-term">P/E (Price/Earnings)</div><div class="gloss-def">Giá cổ phiếu / EPS. Đo lường giá cổ phiếu so với lợi nhuận. P/E thấp = rẻ hơn (thường vậy).</div></div>
      <div class="gloss-item"><div class="gloss-term">P/B (Price/Book)</div><div class="gloss-def">Giá cổ phiếu / Vốn chủ sở hữu trên cổ phiếu. Đo lường giá so với tài sản ròng.</div></div>
      <div class="gloss-item"><div class="gloss-term">EPS</div><div class="gloss-def">Lợi nhuận trên mỗi cổ phiếu (VND). Tổng lợi nhuận / số cổ phiếu.</div></div>
      <div class="gloss-item"><div class="gloss-term">ROE</div><div class="gloss-def">Lợi nhuận / Vốn chủ sở hữu. Hiệu suất sinh lời trên vốn cổ đông.</div></div>
      <div class="gloss-item"><div class="gloss-term">Capex</div><div class="gloss-def">Chi phí vốn — tiền đầu tư vào tài sản dài hạn (cửa hàng, thiết bị).</div></div>
      <div class="gloss-item"><div class="gloss-term">CFO</div><div class="gloss-def">Dòng tiền từ hoạt động kinh doanh — tiền thực tế tạo ra từ bán hàng.</div></div>
      <div class="gloss-item"><div class="gloss-term">FCF</div><div class="gloss-def">Dòng tiền tự do = CFO - Capex. Tiền còn lại sau khi đầu tư.</div></div>
      <div class="gloss-item"><div class="gloss-term">Equity</div><div class="gloss-def">Vốn chủ sở hữu — tài sản ròng thuộc về cổ đông.</div></div>
      <div class="gloss-item"><div class="gloss-term">Break-even</div><div class="gloss-def">Điểm hòa vốn — doanh thu vừa đủ bù đắp chi phí, LN = 0.</div></div>
      <div class="gloss-item"><div class="gloss-term">Drawdown</div><div class="gloss-def">Mức giảm từ đỉnh giá cao nhất xuống đáy. Đo lường rủi ro sụt giảm.</div></div>

(function(){
  const nav = document.getElementById('topnav');
  const links = Array.from(document.querySelectorAll('.topnav-link'));
  const progressBar = document.getElementById('progressBar');
  const backTop = document.getElementById('backTop');
  const sections = links.map(l => document.getElementById(l.dataset.target)).filter(Boolean);

  function onScroll(){
    const y = window.scrollY;
    const h = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (h>0 ? (y/h*100) : 0) + '%';
    if (y > 600) backTop.classList.add('visible'); else backTop.classList.remove('visible');
    const navH = nav.offsetHeight + 10;
    let active = sections[0];
    for (const s of sections){
      if (s && (s.getBoundingClientRect().top - navH) <= 0) active = s;
    }
    links.forEach(l => {
      l.classList.toggle('active', active && l.dataset.target === active.id);
    });
    const activeLink = links.find(l => l.classList.contains('active'));
    if (activeLink){
      const navInner = nav.querySelector('.topnav-inner');
      const linkRect = activeLink.getBoundingClientRect();
      const innerRect = navInner.getBoundingClientRect();
      if (linkRect.right > innerRect.right - 20 || linkRect.left < innerRect.left + 20){
        navInner.scrollTo({ left: navInner.scrollLeft + (linkRect.left - innerRect.left - 40), behavior:'smooth' });
      }
    }
  }
  window.addEventListener('scroll', onScroll, {passive:true});
  window.addEventListener('resize', onScroll);
  links.forEach(l => {
    l.addEventListener('click', (e)=>{
      const t = document.getElementById(l.dataset.target);
      if (!t) return;
      e.preventDefault();
      const top = t.getBoundingClientRect().top + window.scrollY - nav.offsetHeight - 8;
      window.scrollTo({ top, behavior:'smooth' });
    });
  });
  backTop.addEventListener('click', ()=> window.scrollTo({top:0, behavior:'smooth'}));
  onScroll();
})();
</script>
