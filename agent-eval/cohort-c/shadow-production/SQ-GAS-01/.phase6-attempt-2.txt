<section id="sec-glossary" class="section">
  <div class="section-title aos"><span class="num">21</span><h2>📖 Hướng dẫn đọc — Giải thích thuật ngữ</h2><div class="divider"></div></div>
  <div class="section-sub">Thuật ngữ tài chính &amp; ngành dùng trong báo cáo</div>
  <div class="glossary-grid aos">
    <div class="gloss-item"><div class="gloss-term">P/E (Price/Earnings)</div><div class="gloss-def">Giá / EPS. P/E 13× nghĩa là nhà đầu tư trả 13 đồng cho mỗi 1 đồng lợi nhuận.</div></div>
    <div class="gloss-item"><div class="gloss-term">P/B (Price/Book)</div><div class="gloss-def">Giá / Vốn CSH trên cổ phiếu. P/B 3.2× nghĩa giá gấp 3.2 lần giá trị sổ sách.</div></div>
    <div class="gloss-item"><div class="gloss-term">ROE</div><div class="gloss-def">Lợi nhuận / Vốn CSH. ROE 25% = 100 đồng vốn tạo ra 25 đồng lợi nhuận/năm.</div></div>
    <div class="gloss-item"><div class="gloss-term">EPS</div><div class="gloss-def">Lợi nhuận trên cổ phiếu. EPS 5,911 VND = mỗi cổ phiếu "kiếm" 5,911 VND/năm.</div></div>
    <div class="gloss-item"><div class="gloss-term">Capex</div><div class="gloss-def">Chi phí vốn (đầu tư tài sản cố định). Capex cao → FCF giảm.</div></div>
    <div class="gloss-item"><div class="gloss-term">FCF</div><div class="gloss-def">Dòng tiền tự do = CFO - Capex. FCF dương → có tiền trả cổ tức/mua lại.</div></div>
    <div class="gloss-item"><div class="gloss-term">LNG</div><div class="gloss-def">Khí tự nhiên hóa lỏng (Liquefied Natural Gas). Nhập khẩu bằng tàu, khí hóa lại để phân phối.</div></div>
    <div class="gloss-item"><div class="gloss-term">CAGR</div><div class="gloss-def">Tốc độ tăng trưởng kép hàng năm. DT CAGR -5.5% = giảm trung bình 5.5%/năm.</div></div>
    <div class="gloss-item"><div class="gloss-term">Drawdown</div><div class="gloss-def">Khoảng giảm từ đỉnh giá. Drawdown -30% = giá giảm 30% từ đỉnh.</div></div>
    <div class="gloss-item"><div class="gloss-term">RSI</div><div class="gloss-def">Chỉ số sức mạnh tương đối. RSI &gt;70 = quá mua, &lt;30 = quá bán.</div></div>
  </div>

(function(){
  const nav = document.getElementById('topnav');
  const links = Array.from(document.querySelectorAll('.topnav-link'));
  const progressBar = document.getElementById('progressBar');
  const backTop = document.getElementById('backTop');
  const sections = links.map(l => document.getElementById(l.dataset.target)).filter(Boolean);

  function onScroll(){
    const y = window.scrollY;
    const h = document.documentElement.scrollHeight - window.innerHeight;
    progressBar.style.width = (h>0 ? (y/h*100) : 0) + '%';
    if (y > 600) backTop.classList.add('visible'); else backTop.classList.remove('visible');
    const navH = nav.offsetHeight + 10;
    let active = sections[0];
    for (const s of sections){
      if (s && (s.getBoundingClientRect().top - navH) <= 0) active = s;
    }
    links.forEach(l => {
      l.classList.toggle('active', active && l.dataset.target === active.id);
    });
    const activeLink = links.find(l => l.classList.contains('active'));
    if (activeLink){
      const navInner = nav.querySelector('.topnav-inner');
      const linkRect = activeLink.getBoundingClientRect();
      const innerRect = navInner.getBoundingClientRect();
      if (linkRect.right > innerRect.right - 20 || linkRect.left < innerRect.left + 20){
        navInner.scrollTo({ left: navInner.scrollLeft + (linkRect.left - innerRect.left - 40), behavior:'smooth' });
      }
    }
  }
  window.addEventListener('scroll', onScroll, {passive:true});
  window.addEventListener('resize', onScroll);
  links.forEach(l => {
    l.addEventListener('click', (e)=>{
      const t = document.getElementById(l.dataset.target);
      if (!t) return;
      e.preventDefault();
      const top = t.getBoundingClientRect().top + window.scrollY - nav.offsetHeight - 8;
      window.scrollTo({ top, behavior:'smooth' });
    });
  });
  backTop.addEventListener('click', ()=> window.scrollTo({top:0, behavior:'smooth'}));
  onScroll();
})();
